\chapter{Triển khai cài đặt cơ sở dữ liệu}

\newthought{Triển khai cài đặt cơ sở dữ liệu} là giai đoạn chuyển đổi lược đồ thiết kế thành các bảng, index, view và trigger thực tế. Phần này trình bày chi tiết cấu trúc từng bảng, các ràng buộc, index tối ưu hóa, view để lấy dữ liệu và trigger tự động.

\section{Account}

\subsection{Cấu trúc bảng}

Bảng \texttt{accounts} lưu trữ thông tin tài khoản hệ thống.

\begin{verbatim}
CREATE TABLE accounts (
    account_id INT PRIMARY KEY AUTO_INCREMENT,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    role ENUM('buyer', 'seller', 'admin') NOT NULL,
    status ENUM('active', 'suspended', 'deleted') DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP 
        ON UPDATE CURRENT_TIMESTAMP
);
\end{verbatim}

\subsection{Index}

\begin{verbatim}
CREATE INDEX idx_accounts_email ON accounts(email);
CREATE INDEX idx_accounts_role ON accounts(role);
CREATE INDEX idx_accounts_status ON accounts(status);
\end{verbatim}

\subsection{View}

\begin{verbatim}
CREATE VIEW vw_active_accounts AS
SELECT account_id, email, role, created_at
FROM accounts
WHERE status = 'active';
\end{verbatim}

\subsection{Trigger}

\begin{verbatim}
DELIMITER $$
CREATE TRIGGER trg_account_audit
AFTER UPDATE ON accounts
FOR EACH ROW
BEGIN
    INSERT INTO account_audit_log 
    (account_id, old_status, new_status, changed_at)
    VALUES (NEW.account_id, OLD.status, NEW.status, NOW());
END$$
DELIMITER ;
\end{verbatim}

\underline{trigger trg\_after\_insert\_variant:}

\begin{verbatim}
CREATE TRIGGER trg_after_insert_variant
AFTER INSERT ON variants
FOR EACH ROW
BEGIN
    UPDATE products
    SET total_quantity = total_quantity + NEW.quantity
    WHERE id = NEW.id;
END;
\end{verbatim}

Mục đích: tăng total\_quantity trong products khi thêm sản phẩm mới (variant).

\underline{trigger trg\_after\_insert\_order\_item\_sold:}

\begin{verbatim}
CREATE TRIGGER trg_after_insert_order_item_sold
AFTER INSERT ON order_items
FOR EACH ROW
BEGIN
    UPDATE products p
    JOIN variants v ON p.id = v.product_id
    SET p.total_quantity_sold = p.total_quantity_sold + NEW.quantity
    WHERE v.id = NEW.variant_id;
END;
\end{verbatim}

Mục đích: cập nhật total\_quantity\_sold khi bán sản phẩm.

\underline{trigger trg\_after\_delete\_order\_item\_sold:}

\begin{verbatim}
CREATE TRIGGER trg_after_delete_order_item_sold
AFTER DELETE ON order_items

BEGIN
    UPDATE products p
    JOIN variants v ON p.id = v.product_id
    SET p.total_quantity_sold = p.total_quantity_sold - OLD.quantity
    WHERE v.id = OLD.variant_id;
END;
\end{verbatim}

Mục đích: cập nhật total\_quantity\_sold khi hoàn hàng.

CREATE TRIGGER trg_variant_update_delta
AFTER UPDATE ON variants
FOR EACH ROW

    DECLARE b INT;

    SET a = NEW.quantity - OLD.quantity;
    SET b = NEW.quantity_sold - OLD.quantity_sold;

    UPDATE products
    SET total_quantity = total_quantity + a,
        total_quantity_sold = total_quantity_sold + b
    WHERE id = NEW.product_id;
END;

\begin{verbatim}
DELIMITER $$

CREATE PROCEDURE sp_create_order(
    IN p_user_id INT,
    IN p_variant_id INT,
    IN p_quantity INT
)
BEGIN
    DECLARE oid INT;

    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
    END;

    START TRANSACTION;

    INSERT INTO orders(user_id)
    VALUES (p_user_id);

    SET oid = LAST_INSERT_ID();

    INSERT INTO order_items(order_id, variant_id, quantity)
    VALUES (oid, p_variant_id, p_quantity);

    COMMIT;
END$$

DELIMITER ;
\end{verbatim}

Mục đích: tạo đơn hàng với transaction, đảm bảo nếu có lỗi thì rollback toàn bộ.

\underline{function fn\_hot\_variant\_by\_product:}

\begin{verbatim}
DELIMITER $$

CREATE FUNCTION fn_hot_variant_by_product(p_product_id INT)
RETURNS INT
DETERMINISTIC
BEGIN
    DECLARE v_id INT;

    SELECT v.id
    INTO v_id
    FROM order_items oi
    JOIN variants v ON oi.variant_id = v.id
    WHERE v.product_id = p_product_id
      AND v.deleted = 0
    GROUP BY v.id
    ORDER BY SUM(oi.quantity) DESC
    LIMIT 1;

    RETURN v_id;
END$$

DELIMITER ;
\end{verbatim}

Mục đích: trả về id của variant bán chạy nhất (hot) trong các variants của một product.

\underline{procedure sp\_update\_product\_daily:}

\begin{verbatim}
DELIMITER $$

CREATE PROCEDURE sp_update_product_daily(IN p_date DATE)
BEGIN
    DECLARE v_revenue DECIMAL(12,2);
    DECLARE v_orders INT;

    -- Tổng doanh thu trong ngày
    SELECT IFNULL(SUM(oi.quantity * v.price), 0)
    INTO v_revenue
    FROM orders o
    JOIN order_items oi ON o.id = oi.order_id
    JOIN variants v ON oi.variant_id = v.id
    WHERE DATE(o.created_at) = p_date;

    -- Tổng số đơn trong ngày
    SELECT COUNT(*)
    INTO v_orders
    FROM orders
    WHERE DATE(created_at) = p_date;

    -- Cập nhật bảng product_daily
    UPDATE product_daily
    SET total_revenue = v_revenue,
        total_orders = v_orders
    WHERE date = p_date;
END$$

DELIMITER ;
\end{verbatim}

Mục đích: tính tổng doanh thu và số lượng đơn hàng trong một ngày và cập nhật vào bảng product\_daily.

\subsection{Trigger}


\noindent\textbf{trg\_notifications\_before\_insert} --- kiểm tra và set id, set created\_at, khởi tạo is\_read = 0 và read\_at = NULL.

\begin{verbatim}
DELIMITER $$
CREATE TRIGGER trg_notifications_before_insert
BEFORE INSERT ON notifications
FOR EACH ROW
BEGIN
    IF NEW.id IS NULL OR NEW.id = '' THEN
        SET NEW.id = UUID();
    END IF;
    IF NEW.created_at IS NULL THEN
        SET NEW.created_at = NOW(6);
    END IF;
    SET NEW.is_read = 0;
    SET NEW.read_at  = NULL;
END$$
DELIMITER ;
\end{verbatim}

\noindent\textbf{trg\_notifications\_before\_update} --- cập nhật read\_at khi is\_read chuyển từ 0 sang 1.

\begin{verbatim}
DELIMITER $$
CREATE TRIGGER trg_notifications_before_update
BEFORE UPDATE ON notifications
FOR EACH ROW
BEGIN
    IF OLD.is_read = 0 AND NEW.is_read = 1 THEN
        SET NEW.read_at = NOW(6);
    END IF;
END$$
DELIMITER ;
\end{verbatim}

CREATE VIEW cart_details_view AS
SELECT
    sci.id                          AS item_id,
    sc.client_id,
    sci.variant_id,
    sci.quantity,
    sci.price_each,
    (sci.quantity * sci.price_each) AS item_total_price,
    v.thumbnail,
    p.name                          AS product_name

  AND      sc.deleted  = 0;
