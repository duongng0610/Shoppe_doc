\chapter{Phân tích hệ thống}

\newthought{Phân tích hệ thống} là giai đoạn quan trọng giúp xác định rõ các yêu cầu, mục tiêu và các yếu tố tác động của hệ thống e-commerce. Phần này trình bày chi tiết phân tích các khía cạnh của hệ thống cơ sở dữ liệu.

\section{Mục tiêu của hệ thống}

Mục tiêu chính của hệ thống cơ sở dữ liệu e-commerce bao gồm:

\begin{itemize}
    \item Lưu trữ hiệu quả thông tin sản phẩm, người dùng và đơn hàng
    \item Hỗ trợ giao dịch an toàn và tin cậy
    \item Cung cấp hiệu suất truy vấn cao cho các hoạt động thời gian thực
    \item Đảm bảo tính toàn vẹn và nhất quán dữ liệu
    \item Hỗ trợ phân tích dữ liệu chi tiết cho ba nhóm người dùng: Khách hàng, Người bán và Quản trị viên
    \item Cung cấp cơ sở cho các chức năng báo cáo và dự báo
\end{itemize}

\section{Các tác nhân của hệ thống}

Hệ thống phục vụ ba nhóm tác nhân chính:

\subsection{Khách hàng (Buyer/Customer)}
\begin{itemize}
    \item Người dùng chưa đăng nhập (Guest)
    \item Người dùng đã đăng nhập với vai trò mua hàng
\end{itemize}

\subsection{Người bán (Seller/Seller-Account)}
\begin{itemize}
    \item Người bán hàng (Shop Owner)
    \item Quản lý cửa hàng (Shop Manager)
\end{itemize}

\subsection{Quản trị viên (Administrator)}
\begin{itemize}
    \item Quản lý người dùng
    \item Quản lý giao dịch
    \item Phân tích dữ liệu toàn hệ thống
\end{itemize}

\section{Yêu cầu đối với hệ thống}

\subsection{Yêu cầu chức năng}

Hệ thống phải hỗ trợ:
\begin{itemize}
    \item Quản lý tài khoản (Account Management)
    \item Quản lý người dùng (User Profile Management)
    \item Quản lý sản phẩm (Product Management)
    \item Quản lý danh mục (Category Management)
    \item Quản lý đơn hàng (Order Management)
    \item Quản lý giao dịch (Transaction Management)
    \item Báo cáo và phân tích dữ liệu
\end{itemize}

\subsection{Yêu cầu phi chức năng}

Hệ thống phải đạt được:
\begin{itemize}
    \item \textbf{Hiệu suất:} Phản hồi nhanh cho các truy vấn thường gặp
    \item \textbf{Tính mở rộng:} Xử lý tăng thêm dữ liệu và lưu lượng người dùng
    \item \textbf{Tính toàn vẹn:} Đảm bảo dữ liệu luôn nhất quán và chính xác
    \item \textbf{Bảo mật:} Bảo vệ dữ liệu nhạy cảm của người dùng
    \item \textbf{Độ tin cậy:} Giảm thiểu thời gian chết hệ thống
    \item \textbf{Khả năng bảo trì:} Dễ dàng sửa chữa, mở rộng và nâng cấp
\end{itemize}
